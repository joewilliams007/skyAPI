## database
DROP DATABASE db_sky;
CREATE DATABASE db_sky;
ALTER DATABASE db_sky CHARACTER SET utf8mb4 COLLATE utf8mb4_bin;
USE db_sky;

CREATE TABLE Users (
user_id INT NOT NULL AUTO_INCREMENT,
avatar TEXT,
username varchar(255),
color varchar(255),
used_score int DEFAULT 0,
reactions int DEFAULT 0,
blocked_users TEXT,
blocked_words TEXT,
following TEXT,
background varchar(2000),
timestamp int,
PRIMARY KEY ( user_id )
);

ALTER TABLE Users ADD avatar_frame_url VARCHAR(2000) DEFAULT "";
ALTER TABLE Users ADD avatar_bg_hex VARCHAR(255) DEFAULT "";
ALTER TABLE Users ADD avatar_bg_url VARCHAR(2000) DEFAULT "";
ALTER TABLE Users ADD profile_bg_url VARCHAR(2000) DEFAULT "";

CREATE TABLE Auth (
auth_id INT NOT NULL AUTO_INCREMENT,
session_id varchar(255),
user_id int,
verify_key int,
verified BOOLEAN DEFAULT false,
PRIMARY KEY ( auth_id ),
FOREIGN KEY ( user_id )
    REFERENCES Users ( user_id )
    ON DELETE CASCADE
) ENGINE = INNODB; 

CREATE TABLE Reaction (
reaction_id INT NOT NULL AUTO_INCREMENT,
user_id int,
post_id int,
reaction varchar(255),
timestamp int,
PRIMARY KEY ( reaction_id ),
FOREIGN KEY ( user_id )
    REFERENCES Users ( user_id )
    ON DELETE CASCADE
) ENGINE = INNODB; 

CREATE TABLE Projects (
project_id INT NOT NULL AUTO_INCREMENT,
user_id int,
stars int DEFAULT 0,
comments int DEFAULT 0,
reactions int DEFAULT 0,
title varchar(2000),
os varchar(2000),
type varchar(2000),
timestamp_added int,
timestamp_created int,
description TEXT,
relevant_dr_url varchar(2000),
website varchar(2000),
github varchar(2000),
language varchar(2000),
active BOOLEAN,
archived BOOLEAN,
owner_user_id int,
owner varchar(2000),
PRIMARY KEY ( project_id ),
FOREIGN KEY ( user_id )
    REFERENCES Users ( user_id )
    ON DELETE SET NULL
) ENGINE = INNODB;

CREATE TABLE Messages (
message_id INT NOT NULL AUTO_INCREMENT,
user_id int,
body TEXT,
image varchar(2000),
timestamp int,
PRIMARY KEY ( message_id ),
FOREIGN KEY ( user_id )
    REFERENCES Users ( user_id )
    ON DELETE CASCADE
) ENGINE = INNODB;
